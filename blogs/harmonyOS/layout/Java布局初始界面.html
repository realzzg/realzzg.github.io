# 详细代码

package com.example.myapplication.slice;

import ohos.aafwk.ability.AbilitySlice;
import ohos.aafwk.content.Intent;
import ohos.agp.colors.RgbColor;
import ohos.agp.components.*;
import ohos.agp.components.element.ShapeElement;
import ohos.agp.utils.Color;
import ohos.agp.utils.LayoutAlignment;

//SecondSlice 继承于 AbilitySlice
public class SecondSlice extends AbilitySlice {

    //重写onStart();

    @Override
    protected void onStart(Intent intent) {
        super.onStart(intent);

        //创建布局
        DependentLayout layout = new DependentLayout(this);
        //设置布局大小 和父组件一个大小———铺满屏幕
        layout.setWidth(DependentLayout.LayoutConfig.MATCH_PARENT);
        layout.setHeight(DependentLayout.LayoutConfig.MATCH_PARENT);
        //设置组件局中，可选水平居中、垂直居中、居中
        layout.setAlignment(LayoutAlignment.HORIZONTAL_CENTER);
        //新建背景
        ShapeElement background = new ShapeElement();
        //设置布局颜色：白色
        background.setRgbColors(new RgbColor[]{new RgbColor(255, 255, 255)});
        //将布局设置为当前布局
        layout.setBackground(background);

        //新建文本组件
        Text text = new Text(this);
        //设置组件宽度 高度 颜色 文字 大小
        text.setText("你好 世界");
        text.setTextColor(Color.BLACK);
        text.setTextSize(100);

        //与下面注释掉的代码是一个效果
        text.setWidth(DependentLayout.LayoutConfig.MATCH_CONTENT);
        text.setHeight(DependentLayout.LayoutConfig.MATCH_CONTENT);

        DependentLayout.LayoutConfig textConfig = new DependentLayout.LayoutConfig();
        // DependentLayout.LayoutConfig textConfig = new DependentLayout.LayoutConfig(DependentLayout.LayoutConfig.MATCH_CONTENT, DependentLayout.LayoutConfig.MATCH_CONTENT);
        //文字在父容器居中
        textConfig.addRule(DependentLayout.LayoutConfig.CENTER_IN_PARENT);
        text.setLayoutConfig(textConfig);
        //合写报错 DependentLayout.LayoutConfig.CENTER_IN_PARENT是int类型，等于15
        //text.setLayoutConfig(new DependentLayout.LayoutConfig().addRule(15););


        //给布局添加文字组件
        layout.addComponent(text);

        //添加Button组件
        Button button = new Button(getContext());
        //设置组件大小
        button.setWidth(ComponentContainer.LayoutConfig.MATCH_CONTENT);
        button.setHeight(ComponentContainer.LayoutConfig.MATCH_CONTENT);
        //设置组件属性
        button.setText("Next");
        button.setTextSize(60);
        //父组件已设置居中方式

        //设置背景
        ShapeElement backgroundButton = new ShapeElement();
        //和文字的颜色设置不一样
        backgroundButton.setRgbColor(new RgbColor(0, 125, 255));
        //设置圆角
        backgroundButton.setCornerRadius(100);
        //赋给按钮背景
        button.setBackground(backgroundButton);
        //内边距
        button.setPadding(175, 20, 175, 20);

        DependentLayout.LayoutConfig buttonConfig = new DependentLayout.LayoutConfig();
        //在父组件中，居中
        buttonConfig.addRule(DependentLayout.LayoutConfig.HORIZONTAL_CENTER);
        //对齐父组件的底部
        buttonConfig.addRule(DependentLayout.LayoutConfig.ALIGN_PARENT_BOTTOM);
        //设置底部外边距
        buttonConfig.setMarginBottom(40);
        button.setLayoutConfig(buttonConfig);
        layout.addComponent(button);

        //布局添加到组件树
        super.setUIContent(layout);

    }
}
